  <h2>Calculadora de Aumento por Arancel</h2>

  <input type="number" id="precio" placeholder="Ej: 100" min="0" step="0.01">


  <label for="arancel">Porcentaje de arancel:</label>

  <select id="arancel">

    <option value="10">10%</option>

    <option value="125">125%</option>

  </select>


  <button onclick="calcular()">Calcular nuevo precio</button>


  <div id="resultado"></div>


  <script>

    function calcular() {

      var precio = parseFloat(document.getElementById('precio').value);

      var arancel = parseFloat(document.getElementById('arancel').value);

      var resultado = document.getElementById('resultado');


      if (isNaN(precio) || precio <= 0) {

        resultado.innerHTML = "Por favor, ingresa un precio válido.";

        return;

      }


      var aumento = (precio * arancel) / 100;

      var nuevoPrecio = precio + aumento;


      resultado.innerHTML = `

        Aumento: $${aumento.toFixed(2)}<br>

        Nuevo precio: $${nuevoPrecio.toFixed(2)}

      `;

    }

  </script>
